   110→/* ── Relative Time Formatter ── */
   111→function formatRelativeTime(dateStr: string): string {
   112→  try {
   113→    const date = new Date(dateStr);
   114→    const now = new Date();
   115→    const diffMs = now.getTime() - date.getTime();
   116→    const diffSec = Math.floor(diffMs / 1000);
   117→    const diffMin = Math.floor(diffSec / 60);
   118→    const diffHr = Math.floor(diffMin / 60);
   119→    const diffDay = Math.floor(diffHr / 24);
   120→
   121→    if (diffSec < 60) return "Just now";
   122→    if (diffMin < 60) return `${diffMin} min. ago`;
   123→    if (diffHr < 24) return `${diffHr} hr. ago`;
   124→    if (diffDay === 1) return "1 day ago";
   125→    if (diffDay < 30) return `${diffDay} days ago`;
   126→    return date.toLocaleDateString();
   127→  } catch {
   128→    return dateStr;
   129→  }
   130→}

   132→/* ── Recently Active Helper ── */
   133→function isRecentlyActive(dateStr: string): boolean {
   134→  try {
   135→    const date = new Date(dateStr);
   136→    const now = new Date();
   137→    const diffMs = now.getTime() - date.getTime();
   138→    const diffHr = diffMs / (1000 * 60 * 60);
   139→    return diffHr <= 12;
   140→  } catch {
   141→    return false;
   142→  }
   143→}

   320→function OnboardingFlow({ onComplete }: { onComplete: (user: ForumUser) => void }) {
   321→  const [step, setStep] = useState(0);
   322→  const [direction, setDirection] = useState(1);
   323→  const [displayName, setDisplayName] = useState("");
   324→  const [email, setEmail] = useState("");
   325→  const [bio, setBio] = useState("");
   326→  const [role, setRole] = useState("");
   327→  const [location, setLocation] = useState("");
   328→  const [selectedCommunities, setSelectedCommunities] = useState<string[]>([]);
   329→  const [isSubmitting, setIsSubmitting] = useState(false);
   330→  const [avatarUrl, setAvatarUrl] = useState<string>("");
   331→  const [avatarFile, setAvatarFile] = useState<File | null>(null);
   332→
   333→  const goNext = () => { setDirection(1); setStep((s) => s + 1); };
   334→  const goBack = () => { setDirection(-1); setStep((s) => s - 1); };
   335→
   336→  const toggleCommunity = (name: string) => {
   337→    setSelectedCommunities((prev) =>
   338→      prev.includes(name) ? prev.filter((c) => c !== name) : [...prev, name]
   339→    );
   340→  };
   341→
   342→  const handleComplete = async () => {
   343→    setIsSubmitting(true);
   344→    try {
   345→      const avatarColor = ROLE_COLORS[role] || "#6b7280";
   346→      let finalAvatarUrl = avatarUrl;
   347→
   348→      // Upload avatar file if selected
   349→      if (avatarFile) {
   350→        const formData = new FormData();
   351→        formData.append("file", avatarFile);
   352→        const uploadRes = await fetch("/api/upload", {
   353→          method: "POST",
   354→          body: formData,
   355→        });
   356→        if (uploadRes.ok) {
   357→          const uploadData = await uploadRes.json();
   358→          finalAvatarUrl = uploadData.url;
   359→        }
   360→      }
   361→
   362→      // Create user via API
   363→      const res = await fetch("/api/forum/users", {
   364→        method: "POST",
   365→        headers: { "Content-Type": "application/json" },
   366→        body: JSON.stringify({
   367→          name: displayName,
   368→          email,
   369→          bio: bio || undefined,
   370→          role,
   371→          location: location || undefined,
   372→          communities: selectedCommunities,
   373→          avatarColor,
   374→          avatarUrl: finalAvatarUrl || undefined,
   375→        }),
   376→      });
   377→
   378→      if (!res.ok) {
   379→        const err = await res.json();
   380→        console.error("Failed to create user:", err);
   381→        setIsSubmitting(false);
   382→        return;
   383→      }
   384→
   385→      const user: ForumUser = await res.json();
   386→
   387→      // Store user ID in localStorage
   388→      localStorage.setItem("xcelero_townsquare_user_id", user.id);
   389→
   390→      // Seed the database with sample data
   391→      await fetch("/api/forum/seed", { method: "POST" }).catch(() => {});
   392→
   393→      onComplete(user);
   394→    } catch (err) {
   395→      console.error("Onboarding error:", err);
   396→      setIsSubmitting(false);
   397→    }
   398→  };
   399→
   400→  return (
   401→    <div className="flex flex-col min-h-screen bg-white text-[#111111] font-sans">
   402→      {/* Header bar */}
   403→      <header className="h-[56px] bg-white border-b border-[#111111]/10 px-4 md:px-8 flex items-center justify-between shrink-0">
   404→        <div className="flex items-center gap-2.5">

   406→            <MessageSquare className="w-4 h-4 text-white" />
   407→          </div>
   408→          <span className="text-lg font-display font-medium tracking-tight text-[#111111]">
   409→            Town<span className="text-[#111111]/30 font-normal"> Square</span>
   410→          </span>
   411→        </div>
   412→        {/* Step indicator */}
   413→        <div className="flex items-center gap-2">
   414→          {[0, 1, 2, 3].map((i) => (
   415→            <div
   416→              key={i}
   417→              className={`h-1 rounded-full transition-all duration-300 ${
   418→                i === step ? "w-8 bg-[#FF4D00]" : i < step ? "w-4 bg-[#FF4D00]/40" : "w-4 bg-[#111111]/10"
   419→              }`}
   420→            />
   421→          ))}
   422→        </div>
   423→        <div className="w-[100px]" />
   424→      </header>

   426→      {/* Step content */}
