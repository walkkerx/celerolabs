   494→/* Live countdown hook — ticks every second toward a target date */
   495→function useCountdown(targetDate: Date) {
   496→  const [tick, setTick] = useState(0);
   497→  const mountedRef = useRef(false);
   498→
   499→  useEffect(() => {
   500→    mountedRef.current = true;
   501→    const id = setInterval(() => setTick((t) => t + 1), 1000);
   502→    return () => clearInterval(id);
   503→  }, []);
   504→
   505→  const now = mountedRef.current ? Date.now() : targetDate.getTime();
   506→  const diff = Math.max(0, targetDate.getTime() - now);
   507→  const days = Math.floor(diff / 86400000);
   508→  const hours = Math.floor((diff % 86400000) / 3600000);
   509→  const minutes = Math.floor((diff % 3600000) / 60000);
   510→  const seconds = Math.floor((diff % 60000) / 1000);
   511→
   512→  return { days, hours, minutes, seconds, elapsed: diff === 0, mounted: mountedRef.current };
   513→}

   515→/* Pad a number to 2 digits */
   516→function pad2(n: number) {
   517→  return String(n).padStart(2, "0");
   518→}

   705→   NUMBERS SECTION: Animated counting stats
   706→   ══════════════════════════════════════════════════════════════════════════ */
   707→function AnimatedCounter({ value, prefix, suffix, duration = 2 }: {
   708→  value: number;
   709→  prefix: string;
   710→  suffix: string;
   711→  duration?: number;
   712→}) {
   713→  const [count, setCount] = useState(0);
   714→  const [hasAnimated, setHasAnimated] = useState(false);
   715→  const ref = useRef<HTMLDivElement>(null);
   716→
   717→  useEffect(() => {
   718→    const el = ref.current;
   719→    if (!el || hasAnimated) return;
   720→
   721→    const observer = new IntersectionObserver(
   722→      ([entry]) => {
   723→        if (entry.isIntersecting && !hasAnimated) {
   724→          setHasAnimated(true);
   725→          observer.disconnect();
   726→
   727→          let startTime: number | null = null;
   728→          let animationFrame: number;
   729→
   730→          const animate = (timestamp: number) => {
   731→            if (!startTime) startTime = timestamp;
   732→            const progress = Math.min((timestamp - startTime) / (duration * 1000), 1);
   733→            const eased = 1 - Math.pow(1 - progress, 3);
   734→            setCount(value * eased);
   735→
   736→            if (progress < 1) {
   737→              animationFrame = requestAnimationFrame(animate);
   738→            }
   739→          };
   740→
   741→          animationFrame = requestAnimationFrame(animate);
   742→          return () => cancelAnimationFrame(animationFrame);
   743→        }
   744→      },
   745→      { threshold: 0.2 }
   746→    );
   747→
   748→    observer.observe(el);
   749→    return () => observer.disconnect();
   750→  }, [hasAnimated, value, duration]);
   751→
   752→  const displayValue = value % 1 === 0 ? Math.round(count) : count.toFixed(1);
   753→
   754→  return (
   755→    <div ref={ref} className="tabular-nums">
   756→      <span className="text-[36px] sm:text-[48px] md:text-[72px] lg:text-[88px] font-display font-medium tracking-[-0.03em] leading-[1]">
   757→        {prefix}{displayValue}{suffix}
   758→      </span>
   759→    </div>
   760→  );
   761→}
